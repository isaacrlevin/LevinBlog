{{- /* Simple replacement for the removed built-in `gist` shortcode. Usage: {{< gist user id [file] >}} */ -}} {{- $user := .Get 0 -}} {{- $id := .Get 1 -}} {{- $file := .Get 2 | default "" -}} {{- if and $user $id -}} {{- $src := printf "https://gist.github.com/%s/%s.js" $user $id -}} {{- if ne $file "" -}} {{- $src = printf "%s?file=%s" $src $file -}} {{- end -}} {{- else -}} {{- errorf "gist shortcode requires at least user and id" -}} {{- end -}}
